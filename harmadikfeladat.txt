<script>

//Harmadik feladat

let alsohatar=prompt("Add meg az intervallum alsó határát!");
let felsohatar=prompt("Add meg az intervallum felső határát!");

let szam=Math.round((Math.random()*(felsohatar-alsohatar))+alsohatar);
if (szam%2==0){
	document.write(`${szam}`);
    }
else{
	szam++
	document.write(`${szam}`);
	}


</script>
